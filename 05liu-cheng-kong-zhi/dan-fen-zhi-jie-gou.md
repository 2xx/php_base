#### 单分支

```php
<?php

    /* 语法格式:
        if (条件判断) {
            // 条件成立时, 执行这里的代码
        }
    */

    if ( 1 > 2 ) {
        echo '今天不用写作业!';   // 1 > 2 结果为假, 所以这行代码不会执行.
    }

?>
```

#### 双分支

```php
<?php

    /* 语法格式:
        if (条件判断) {
            // 条件成立时, 执行这里的代码
        } else {
            // 条件不成立, 执行这里的代码
        }
    */

    if (1 > 2) {
        echo '代码敲3遍';
    } else {
        echo '代码手写';
    }


?>
```

#### 多分支1

```php
<?php

    /* 语法格式:
        if (条件1) {
            // 条件1成立时, 执行这里的代码
        } else if (条件2) {   // 条件1不成立时,才会判断条件2
            ... ...
        } else if (条件3) {
            ... ...
        } else if (条件n) {
            ... ...
        } else {
            // 以上条件都不满足时, 执行这里
        }
    */

    if ($score >= 90 && $score <=100) {
        echo '优秀!';
    } else if ($score >= 80 && $score < 90) {
        echo '良好';
    } else if ($score >= 70 && $score < 80) {
        echo '中等';
    } else if ($score >= 60 && $score < 70) {
        echo '及格';
    } else if ($score >= 0 && $score < 60) {
        echo '不及格!';
    } else {
        echo '成绩不合法';
    }

    // 这种结构, 之前的条件成立,就不会再判断之后的条件, 所以大部分 &&后面的代码可以剩略
?>
```
